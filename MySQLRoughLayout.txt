Tables

	Students
			studentid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			fname VARCHAR(40) NOT NULL
			lname VARCHAR(40) NOT NULL
			email VARCHAR(100) NOT NULL
			gradelevel TINYINT UNSIGNED NOT NULL
			
	StudentsClasses
			studentid MEDIUMINT NOT NULL PRIMARY KEY
			periodid MEDIUMINT NOT NULL PRIMARY KEY
			classid MEDIUMINT NOT NULL PRIMARY KEY
			
	Teachers
			teacherid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			fname VARCHAR(40) NOT NULL
			lname VARCHAR(40) NOT NULL
			email VARCHAR(100) NOT NULL
			
	Classes
			classid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			teacherid MEDIUMINT NOT NULL
			classname VARCHAR(50) NOT NULL
			
	Grades
			gradeid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			assignmentid MEDIUMINT NOT NULL
			classid MEDIUMINT NOT NULL
			studentid MEDIUMINT NOT NULL
			gradevalue VARCHAR(3) NOT NULL
			
	Attendances
			datetaken DATE NOT NULL PRIMARY KEY
			studentid MEDIUMINT NOT NULL PRIMARY KEY
			classid MEDIUMINT NOT NULL PRIMARY KEY
			attendancevalue TINYINT UNSIGNED NOT NULL
			
	AssignmentCategories
			categoryid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			classid MEDIUMINT NOT NULL
			categoryname VARCHAR(50) NOT NULL
			categoryweight TINYINT UNSIGNED NOT NULL
			
	Assignments
			assignmentid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			classid MEDIUMINT NOT NULL
			assignmentname VARCHAR(100) NOT NULL
			categoryid MEDIUMINT NOT NULL
			
	Periods
			periodid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY
			periodname VARCHAR(20) NOT NULL
			periodnumber TINYINT UNSIGNED NOT NULL
			
			
			
			
			
CREATE TABLE IF NOT EXISTS students
(
studentid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY,
fname VARCHAR(50) NOT NULL,
lname VARCHAR(50) NOT NULL,
email VARCHAR(150) NOT NULL,
gradelevel TINYINT UNSIGNED NOT NULL
);

CREATE TABLE IF NOT EXISTS teachers
(
teacherid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY,
fname VARCHAR(50) NOT NULL,
lname VARCHAR(50) NOT NULL,
email VARCHAR(150) NOT NULL
);

CREATE TABLE IF NOT EXISTS classes
(
classid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY,
teacherid MEDIUMINT NOT NULL,
classname VARCHAR(50) NOT NULL
);

CREATE TABLE IF NOT EXISTS grades
(
gradeid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY,
assignmentid MEDIUMINT NOT NULL,
classid MEDIUMINT NOT NULL,
studentid MEDIUMINT NOT NULL,
gradevalue VARCHAR(3) NOT NULL
);

CREATE TABLE IF NOT EXISTS attendances
(
datetaken DATE NOT NULL,
studentid MEDIUMINT NOT NULL,
classid MEDIUMINT NOT NULL,
attendancevalue TINYINT UNSIGNED NOT NULL,
PRIMARY KEY(datetaken, studentid, classid)
);

CREATE TABLE IF NOT EXISTS assignmentcategories
(
categoryid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL,
classid MEDIUMINT NOT NULL,
categoryname VARCHAR(50) NOT NULL,
categoryweight TINYINT UNSIGNED NOT NULL
);

CREATE TABLE IF NOT EXISTS assignments
(
assignmentid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY,
classid MEDIUMINT NOT NULL,
assignmentname VARCHAR(100) NOT NULL,
categoryid MEDIUMINT NOT NULL
);

CREATE TABLE IF NOT EXISTS periods
(
periodid MEDIUMINT AUTO_INCREMENT NOT NULL PRIMARY KEY,
periodname VARCHAR(20) NOT NULL,
periodnumber TINYINT UNSIGNED NOT NULL
);

CREATE TABLE IF NOT EXISTS studentsclasses
(
studentid MEDIUMINT NOT NULL,
periodid MEDIUMINT NOT NULL,
classid MEDIUMINT NOT NULL,
PRIMARY KEY(studentid, periodid, classid)
);
