Tables

	Students
			studentid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			fname VARCHAR(40) NOT NULL
			lname VARCHAR(40) NOT NULL
			email VARCHAR(100) NOT NULL
			gradelevel TINYINT UNSIGNED NOT NULL
			
	Teachers
			teacherid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			fname VARCHAR(40) NOT NULL
			lname VARCHAR(40) NOT NULL
			email VARCHAR(100) NOT NULL
			
	Users (Handled by Identity)
			userid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			fname VARCHAR(100) NOT NULL
			lname VARCHAR(100) NOT NULL
			email VARCHAR(100) NOT NULL	
			
	Classes
			classid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			teacherid MEDIUMINT NOT NULL
			classname VARCHAR(50) NOT NULL
			
	Grades
			gradeid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			assignmentid MEDIUMINT NOT NULL
			classid MEDIUMINT NOT NULL
			studentid MEDIUMINT NOT NULL
			gradevalue VARCHAR(3) NOT NULL
			
	Attendances
			datetaken DATE PRIMARY KEY NOT NULL
			studentid MEDIUMINT PRIMARY KEY NOT NULL
			classid MEDIUMINT PRIMARY KEY NOT NULL
			attendancevalue TINYINT UNSIGNED NOT NULL
			
	AssignmentCategories
			categoryid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			classid MEDIUMINT NOT NULL
			categoryname VARCHAR(50) NOT NULL
			categoryweight TINYINT UNSIGNED NOT NULL
			
	Assignments
			assignmentid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL
			classid MEDIUMINT NOT NULL
			assignmentname VARCHAR(100) NOT NULL
			categoryid MEDIUMINT NOT NULL
			
			
			
			
			
CREATE TABLE students
(
studentid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL,
fname VARCHAR(50) NOT NULL,
lname VARCHAR(50) NOT NULL,
email VARCHAR(150) NOT NULL,
gradelevel TINYINT UNSIGNED NOT NULL
);

CREATE TABLE teachers
(
teacherid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL,
fname VARCHAR(50) NOT NULL,
lname VARCHAR(50) NOT NULL,
email VARCHAR(150) NOT NULL
);

CREATE TABLE classes
(
classid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL,
teacherid MEDIUMINT NOT NULL,
classname VARCHAR(50) NOT NULL
);

CREATE TABLE grades
(
gradeid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL,
assignmentid MEDIUMINT NOT NULL,
classid MEDIUMINT NOT NULL,
studentid MEDIUMINT NOT NULL,
gradevalue VARCHAR(3) NOT NULL
);

CREATE TABLE attendances
(
datetaken DATE PRIMARY KEY NOT NULL,
studentid MEDIUMINT PRIMARY KEY NOT NULL,
classid MEDIUMINT PRIMARY KEY NOT NULL,
attendancevalue TINYINT UNSIGNED NOT NULL
);

CREATE TABLE assignmentcategories
(
categoryid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NULL,
classid MEDIUMINT NOT NULL,
categoryname VARCHAR(50) NOT NULL,
categoryweight TINYINT UNSIGNED NOT NULL
);

CREATE TABLE assignments
(
assignmentid MEDIUMINT AUTO_INCREMENT PRIMARY KEY NOT NULL,
classid MEDIUMINT NOT NULL,
assignmentname VARCHAR(100) NOT NULL,
categoryid MEDIUMINT NOT NULL
);
