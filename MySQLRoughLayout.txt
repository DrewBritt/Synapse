Tables

	Students
			studentid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			fname VARCHAR(40) (NOT NULL)
			lname VARCHAR(40) (NOT NULL)
			email VARCHAR(100) (NOT NULL)
			gradelevel TINYINT UNSIGNED (NOT NULL)
			
	Teachers
			teacherid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			fname VARCHAR(40) (NOT NULL)
			lname VARCHAR(40) (NOT NULL)
			email VARCHAR(100) (NOT NULL)
			
	Users
			userid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			fname VARCHAR(100) (NOT NULL)
			lname VARCHAR(100) (NOT NULL)
			email VARCHAR(100) (NOT NULL)	
			
	Classes
			classid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			teacherid MEDIUMINT (NOT NULL)
			
	Grades
			gradeid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			assignmentid MEDIUMINT (NOT NULL)
			classid MEDIUMINT (NOT NULL)
			studentid MEDIUMINT (NOT NULL)
			gradevalue VARCHAR(3) (NOT NULL)
			
	Attendances
			datetaken DATE (PRIMARY KEY) (NOT NULL)
			studentid MEDIUMINT (PRIMARY KEY) (NOT NULL)
			classid MEDIUMINT (PRIMARY KEY) (NOT NULL)
			attendancevalue TINYINT UNSIGNED (NOT NULL)
			
	AssignmentCategories
			categoryid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			classid MEDIUMINT (NOT NULL)
			categoryname VARCHAR(50) (NOT NULL)
			categoryweight TINYINT UNSIGNED (NOT NULL)
			
	Assignments
			assignmentid MEDIUMINT (AUTO INCREMENT) (PRIMARY KEY) (NOT NULL)
			classid MEDIUMINT (NOT NULL)
			assignmentname VARCHAR(100) (NOT NULL)
			categoryid MEDIUMINT (NOT NULL)
